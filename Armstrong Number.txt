import 'dart:math';
void main()
{
  int num=153;
  int temp=num;
  int sum=0;
  int len=num.toString().length;
  while(num>0)
  {
    int digit=num%10;
    sum+=pow(digit,len) as int;
    num=num ~/10;
  }
  if(temp==sum)
  {
    print("$temp is a armstrong number");
  }
  else{
    print("$temp is not a armstrong number");
  }
  
}
